// Function to display products on the homepage
function displayFeaturedProducts() {
    const featuredSection = document.querySelector('.featured-products .product-grid');
    products.forEach(product => {
        const productDiv = document.createElement('div');
        productDiv.classList.add('product');
        productDiv.innerHTML = `
            <img src="${product.image}" alt="${product.name}">
            <h3>${product.name}</h3>
            <p>${product.price}</p>
            <a href="${product.page}">View Details</a>
        `;
        featuredSection.appendChild(productDiv);
    });
}

// Function to display all products on the products page
function displayAllProducts() {
    const productListSection = document.querySelector('.product-list .product-grid');
    products.forEach(product => {
        const productDiv = document.createElement('div');
        productDiv.classList.add('product');
        productDiv.innerHTML = `
            <img src="${product.image}" alt="${product.name}">
            <h3>${product.name}</h3>
            <p>${product.price}</p>
            <a href="${product.page}">View Details</a>
        `;
        productListSection.appendChild(productDiv);
    });
}

// Initialize functions based on the page
if (document.querySelector('.featured-products')) {
    displayFeaturedProducts();
} else if (document.querySelector('.product-list')) {
    displayAllProducts();
}
